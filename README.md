UBUNTU INSTALL.

$ apt update && apt upgrade

$ apt install python3

$ apt install nano wget curl -y

$ apt install git

$ git clone https://github.com/cyberscanners/CyberScanners-UBUNTU-VPS-1.git

$ cd CyberScanners-UBUNTU-VPS-1

$ bash vps.sh
